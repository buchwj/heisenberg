299  template <typename HistoCounter>
300  __device__ __forceinline__ void InitHistogram(HistoCounter histogram[BINS])
301  {
302  // Initialize histogram bin counts to zeros
303  int histo_offset = 0;
304 
305  #pragma unroll
306  for(; histo_offset + BLOCK_THREADS <= BINS; histo_offset += BLOCK_THREADS)
307  {
308  histogram[histo_offset + linear_tid] = 0;
309  }
310  // Finish up with guarded initialization if necessary
311  if ((BINS % BLOCK_THREADS != 0) && (histo_offset + linear_tid < BINS))
312  {
313  histogram[histo_offset + linear_tid] = 0;
314  }
315  }
