164 template <
165  typename T,
166  int ITEMS_PER_THREAD>
167 __device__ __forceinline__ void LoadDirectBlockedVectorized(
168  int linear_tid,
169  T *block_ptr,
170  T (&items)[ITEMS_PER_THREAD])
171 {
172  enum
173  {
174  // Maximum CUDA vector size is 4 elements
175  MAX_VEC_SIZE = CUB_MIN(4, ITEMS_PER_THREAD),
176 
177  // Vector size must be a power of two and an even divisor of the items per thread
178  VEC_SIZE = ((((MAX_VEC_SIZE - 1) & MAX_VEC_SIZE) == 0) && ((ITEMS_PER_THREAD % MAX_VEC_SIZE) == 0)) ?
179  MAX_VEC_SIZE :
180  1,
181 
182  VECTORS_PER_THREAD = ITEMS_PER_THREAD / VEC_SIZE,
183  };
184 
185  // Vector type
186  typedef typename CubVector<T, VEC_SIZE>::Type Vector;
187 
188  // Vector items
189  Vector vec_items[VECTORS_PER_THREAD];
190 
191  // Aliased input ptr
192  Vector *ptr = reinterpret_cast<Vector*>(block_ptr + (linear_tid * VEC_SIZE * VECTORS_PER_THREAD));
193 
194  // Load directly in thread-blocked order
195  #pragma unroll
196  for (int ITEM = 0; ITEM < VECTORS_PER_THREAD; ITEM++)
197  {
198  vec_items[ITEM] = ptr[ITEM];
199  }
200 
201  // Copy
202  #pragma unroll
203  for (int ITEM = 0; ITEM < ITEMS_PER_THREAD; ITEM++)
204  {
205  items[ITEM] = reinterpret_cast<T*>(vec_items)[ITEM];
206  }
207 }
